// https://dbdiagram.io/d


Table Groups {
  id bigint [pk]
  bot_id int(12)
  valid varchar(1)
  link varchar(250)
  last_update datetime
  type varchar(250)
  title varchar(250)
}

Table PeopleInEntities {
  id_entity int [ref: > Entities.id]
  id_person int
  
}

Table Entities {
  id int [pk]
  name varchar
}

Table Messages {
  id int [pk]
  from_id_person int [ref: > PeopleInEntities.id_person]
  from_id_entity int [ref: > Entities.id]
  type int [ref: > MessageTypes.id]
  id_photo int [ref: > Photos.id_photo]
  id_video int [ref: > Videos.id_video]
  id_file int [ref: > Files.id_file]
  id_voice int [ref: > Voices.id_voice]
  id_audio int [ref: > Audios.id_audio]
  message_text text
  sent_date datetime
  has_been_sent boolean
  message_id_tg_from int(12)
  message_id_tg_to int(12)
  id_chat_sent_into bigint [ref: > Groups.id]
  from_id_bot int
}

Table MessageTypes {
  id int [pk]
  name varchar(250)
}

Table Photos {
  id_photo int [pk]
  file_id varchar(250)
  file_size int
  height int
  width int
  unique_id varchar(250)
}

Table Videos {
  id_video int [pk]
}

Table Files {
  id_file int [pk]
}

Table Voices {
  id_voice int [pk]
}

Table Audios {
  id_audio int [pk]
}
